81 bool ROSEE::Parser::parseSRDF() {
82 
83  // initialize srdfdom
84  _srdfdom.initFile ( *_urdf_model, _srdf_path );
85 
86  // get the end-effectors in the SRDF file
87  std::vector<srdf::Model::EndEffector> srdf_end_effectors = _srdfdom.getEndEffectors();
88 
89  // NOTE only one end-effectors supported
90  // TBD support multiple end-effectors
91  if ( srdf_end_effectors.size() != 1 ) {
92 
93  return false;
94  }
95 
96  ROS_INFO_STREAM ( "ROSEndEffector Parser found end_effector: " << srdf_end_effectors.at ( 0 ).name_ );
97 
98  // get all the groups in the SRDF
99  std::vector<srdf::Model::Group> srdf_groups = _srdfdom.getGroups();
100 
101  // TBD will be a vector
102  srdf::Model::Group fingers_group;
103  // find end effector fingers chains group
104  // TBD will be a vector
105  int group_num = srdf_groups.size();
106  // TBD will be a vector
107  std::string end_effector_group_name = srdf_end_effectors.at ( 0 ).component_group_;
108  for ( int i = 0; i < group_num; i++ ) {
109  if ( srdf_groups[i].name_ == end_effector_group_name ) {
110  fingers_group = srdf_groups[i];
111  ROS_INFO_STREAM ( "ROSEndEffector Parser found group: " << end_effector_group_name << " in the SRDF with the following fingers: " );
112  }
113  }
114 
115  _fingers_num = fingers_group.subgroups_.size();
116  _fingers_names.resize ( _fingers_num );
117  _fingers_group_id.resize ( _fingers_num );
118 
119  // fill the chain names vector
120  for ( int i = 0; i < _fingers_num; i++ ) {
121 
122  _fingers_names[i] = fingers_group.subgroups_[i];
123 
124  // find the id of the current finger group
125  for ( int j = 0; j < group_num; j++ ) {
126  if ( srdf_groups[j].name_ == _fingers_names[i] ) {
127  _fingers_group_id[i] = j;
128  }
129  }
130  ROS_INFO_STREAM ( _fingers_names[i] );
131  }
132 
133  // iterate over the fingers and find revolute joints
134  for ( int i = 0; i < _fingers_num; i++ ) {
135  srdf::Model::Group current_finger_group = srdf_groups[ _fingers_group_id[i] ];
136 
137  ROS_INFO_STREAM ( "Actuated joints in finger: " << current_finger_group.name_ );
138 
139  // NOTE only one chain per group
140  if ( current_finger_group.chains_.size() != CHAIN_PER_GROUP ) {
141 
142  ROS_ERROR_STREAM ( "for the finger chain groups you can specify only one chain per group in your SRDF check " << current_finger_group.name_.c_str() << "group" );
143  return false;
144  }
145 
146  // fill the enabled/disabled joints in chain map
147  if ( !getJointsInFinger ( current_finger_group.chains_[0].first,
148  current_finger_group.chains_[0].second,
149  current_finger_group.name_
150  ) ) {
151  return false;
152  }
153  }
154 
155  return true;
156

162  std::string xml_string;
163  std::fstream xml_file ( _urdf_path.c_str(), std::fstream::in );
164  if ( xml_file.is_open() ) {
165 
166  while ( xml_file.good() ) {
167 
168  std::string line;

170  xml_string += ( line + "\n" );
171  }
172 
173  xml_file.close();
174  _urdf_model = urdf::parseURDF ( xml_string );
175 
176  // create the robot KDL tree from the URDF model
177  if ( !kdl_parser::treeFromUrdfModel ( *_urdf_model, _robot_tree ) ) {
178 
179  ROS_ERROR_STREAM ( "in " << __func__ << " Failed to construct kdl tree" );
180  return false;

182 
183  // save urdf and string (can be useful to have, thanks @alaurenzi!)
184  std::ifstream t_urdf ( _urdf_path );
185  std::stringstream buffer_urdf;
186  buffer_urdf << t_urdf.rdbuf();
187  _urdf_string = buffer_urdf.str();
188 
189  return true;
190 
191  } else {
192 
193  ROS_ERROR_STREAM ( "in " << __func__ << " : Can NOT open " << _urdf_path << " !" );
